//  Good
#include <stdio.h>
int main(void) {
    // Decentralized processing.
    int num[10] = {1,2,3,4,5,6,7,8,9,0};
    int numWidth = sizeof(num) / sizeof(num[0]);
    int count,countAgain;
    for (count = 0;count < numWidth;count ++) {
        for (countAgain = count+1;countAgain < numWidth;countAgain ++) { 
            if (num[count] < num[countAgain]) {
                int save = num[count];
                num[count] = num[countAgain];
                num[countAgain] = save;
            }
        }
    }
    
    for (count = 0;count < numWidth;count ++) {
        printf("%d", num[count]);
    }
    
    return 0;
}
