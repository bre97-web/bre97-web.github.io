What's Perfect Code_Style ?

名称是鲜活的,它应该拥有意义.
它指引我们,目标明确,我们可以安心跟随它.
int main(void) {
  printf("Hello World !\n");

  return 0;
}
上面的名称,例如int,void等,让我们知道main的参数,我们能清晰记住它.

风格不分对错,你可能注意到了,上面实例使用的是 2 space.
int main(void) {
  printf("2空格 和 4空格 , 各有优劣");
    printf("可为难了Tab");

  return 0;
}
为了代码统一缩进,我们建议使用4空格(传统设定).

统一的代码,必定统一代码格式,我们推荐以下写法
//  函数注释应该在函数上方,不应该在函数内部.
//  所有单行注释前应该添加2空格,且单独成立,不允许和代码部分同行.
int mian(void) {
    //  优秀的代码是高性能的,是极度简洁的.
    int count;

    //  代码之间应该适度留白.   
    for(count = 0;count < 2;count ++) {
        printf("%d", count);
    }

    //  缩进是必不可少的,switch中每一个case(包括default)都应该缩进.
    switch(count) {
        case 1:
            printf("A");
        case 2:
            printf("B")'
        default:
            printf("End");
            break;
    }

    //  最后一行不留空位.
    return 0;
}

很多初学者都会在switch和if中犯错,示例:
//  不规范示范部分,不规范代码上方是提示.
//  ) {之间应该拥有空格.
int main(void){

    //  没有衔接int main(void) {的下一行.
    int age;
    int count;

    //  无论for的执行部分是否一行,都应该添加{ }.
    for(count = 0;count == 0;count ++)
        goto judge;

    //  无论如何,标签都应该考虑缩进.
judge:
    if (count != 2) {
        //  ,的前面不应该空格.
        printf("%d" , count);

    //  } else {应该拥有空格,或者else {移至下一行.
    }else{
        switch(count) {
        //  case应该缩进.case内容应该独立成块.
        case 0:printf("0");
           //  case应该缩进.case内容应该独立成块.
        case 1:printf("1");
        //  default应该缩进.default内容应该独立成块.
        default:break;
        }
    }

    //  最后一行不应该留白.
    return 0;

}
